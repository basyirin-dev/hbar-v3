% --- Definitions and Assumptions ---
\begin{definition}[Parametric Depth Proxy]
Let $\mathcal{D}_{\text{ID}}$ be the in-distribution dataset.
Define the observable proxy for parametric depth as
\[
\delta_A^{\text{obs}}(t) \defeq \text{Acc}_{\text{ID}}(t) \in [0,1],
\]
the accuracy on $\mathcal{D}_{\text{ID}}$. This is the only training-time
observable available to a standard SGD pipeline (the original manuscript
defines $\delta_A$ as a latent variable, but operations
that distinguish $\delta_A$ and $\sigma_A$ require the Stage~1/2 proxy
architecture available only in the $\Sigma$-Model framework).
\end{definition}

\begin{definition}[Schema Coherence Proxy]
Let $\mathcal{D}_{\text{OOD}}$ be an out-of-distribution recombination set
generated by the same grammar $G$. Define
\[
\sigma_A^{\text{obs}}(t) \defeq \frac{\text{Acc}_{\text{OOD}}(t)}{\text{Acc}_{\text{ID}}(t)} \in [0,1].
\]
This matches the Stage~2 ground-truth estimate described in the
original manuscript.
\end{definition}

\begin{assumption}[Compositional Grammar]
The domain $d$ possesses a generative grammar $G(d)$ that defines:
\begin{enumerate}[nosep]
\item A finite set of primitives $\mathcal{P} = \{p_1, \dots, p_k\}$.
\item A set of composition rules $\mathcal{R}$ that combine primitives
  into expressions. Each rule $r \in \mathcal{R}$ has arity $a(r) \in \{1,2\}$.
\item A distribution over expressions $p(e)$ that governs the training
  distribution $\mathcal{D}_{\text{ID}}$.
\item A disjoint OOD distribution $\mathcal{D}_{\text{OOD}}$ over
  expressions that involve novel recombinations of the same primitives.
\end{enumerate}
\end{assumption}

\begin{assumption}[Minimal Expressivity]
The student network $f_\theta$ has sufficient capacity to memorize
the training set $\mathcal{D}_{\text{ID}}$ exactly, and also has sufficient
capacity to implement a compositional solution if trained appropriately.
Formally, there exist weight vectors $\theta_1, \theta_2$ such that:
\[
f_{\theta_1}(x) = f_{\theta_2}(x) = y \quad \forall (x,y) \in \mathcal{D}_{\text{ID}}.
\]
\end{assumption}

% --- Lemma Statement ---
\begin{lemma}[Latentness of Schema Coherence]
\label{lemma:latentness}
Under Assumptions 1--2, the mapping
\[
\delta_A^{\text{obs}} \mapsto \sigma_A^{\text{obs}}
\]
is not injective. Equivalently, there exist weight configurations
$W_1, W_2$ such that:
\[
\delta_A^{\text{obs}}(W_1) = \delta_A^{\text{obs}}(W_2)
\quad\text{but}\quad
\sigma_A^{\text{obs}}(W_1) \neq \sigma_A^{\text{obs}}(W_2).
\]
Therefore, $\sigma_A$ cannot be inferred from $\delta_A$ alone without
additional structural assumptions or measurements.
\end{lemma}

% --- Proof ---
\begin{proof}
We construct explicit weight vectors $W_1, W_2$ and verify the claims.

\paragraph{Network architecture.}
Consider a two-layer network $f: \R^m \to \R$:
\[
f_W(x) = v^\top \phi(U x + b), \qquad
W = \{U \in \R^{h \times m}, v \in \R^h, b \in \R^h\},
\]
with $h \geq |\mathcal{P}| + |\mathcal{R}|$ hidden units and
$\phi(z) = \max(0, z)$ (ReLU). Inputs $x \in \R^m$ are one-hot or
feature-based encodings of grammar expressions.

\paragraph{Training set.}
Let $\mathcal{D}_{\text{ID}}$ consist of $N$ labelled examples
$(x_i, y_i)$ generated by $G(d)$. Since $N \leq h$, the network
can memorise by allocating one hidden unit per training example.

\paragraph{Construction of $W_1$ (surface-statistical configuration).}
Set the first $N$ hidden units as memory slots:
\[
U_{j,\cdot} = x_j^\top, \quad b_j = -\tfrac{1}{2}, \quad
v_j = y_j \quad \text{for } j = 1,\dots,N,
\]
and zero elsewhere. For each training input $x_j$:
\[
\phi(U x_j + b)_j = \phi(x_j^\top x_j - \tfrac{1}{2}) = \|x_j\|^2 - \tfrac{1}{2} > 0,
\]
and $\phi(U x_j + b)_k = 0$ for $k \neq j$. Hence $f_{W_1}(x_j) = y_j$
exactly. For OOD inputs, the network outputs a linear combination of
training labels weighted by input similarity --- a nearest-neighbour
interpolant. In general $f_{W_1}(x_{\text{OOD}}) \neq y_{\text{OOD}}$
when the OOD example is not a convex combination of ID examples.
Thus:
\[
\delta_A^{\text{obs}}(W_1) = \text{Acc}_{\text{ID}} = 1.0, \qquad
\sigma_A^{\text{obs}}(W_1) = \frac{\text{Acc}_{\text{OOD}}}{\text{Acc}_{\text{ID}}} < 1.
\]

\paragraph{Construction of $W_2$ (compositional configuration).}
Suppose each grammar rule $r \in \mathcal{R}$ can be embedded as a
weight matrix $W_r \in \R^{m \times m}$ such that applying $r$
corresponds to $x \mapsto \phi(W_r x)$. Allocate one hidden unit
per primitive for encoding and remaining units for rule composition.
Construct $U, v$ such that the forward pass computes the compositional
parse tree bottom-up. For a given OOD input $x_{\text{OOD}}$ that
recombines known primitives via known rules, $f_{W_2}$ computes
the correct output by compositional generalization. Hence:
\[
\delta_A^{\text{obs}}(W_2) = \text{Acc}_{\text{ID}} = 1.0, \qquad
\sigma_A^{\text{obs}}(W_2) = 1.0.
\]

\paragraph{Verification of non-injectivity.}
Both $W_1$ and $W_2$ achieve $\delta_A^{\text{obs}} = 1$ (perfect ID
accuracy). However:
\[
\sigma_A^{\text{obs}}(W_1) < 1 = \sigma_A^{\text{obs}}(W_2).
\]
Therefore $\delta_A^{\text{obs}}(W_1) = \delta_A^{\text{obs}}(W_2)$
does not imply $\sigma_A^{\text{obs}}(W_1) = \sigma_A^{\text{obs}}(W_2)$.
The mapping is not injective. $\square$
\end{proof}

% --- Boundary Cases ---
\begin{boundary-case}
\textbf{$\delta_A^{\text{obs}} = 0$:} If the agent has acquired no depth,
$\sigma_A^{\text{obs}} = 0/0$ is undefined. The lemma requires
$\text{Acc}_{\text{ID}} > 0$ to establish the counterexample, consistent
with the paper's domain of interest (agents with non-zero depth).

\textbf{$\delta_A^{\text{obs}} = 1$ ceiling:} The construction uses
perfect ID accuracy for clarity. The same argument works for any
$\text{Acc}_{\text{ID}} = c < 1$ as long as both $W_1$ and $W_2$ achieve
the same ID accuracy. The construction generalizes via capacity-accuracy
tradeoff bounds.

\textbf{$|D| = 1$:} The lemma applies per-domain; a single domain is
sufficient for the counterexample.

\textbf{Discontinuity at generation gap:} If the grammar $G(d)$ generates
both ID and OOD sets from the same rule set, the discontinuity in OOD
accuracy between $W_1$ and $W_2$ is not a pathology but the precise
phenomenon the lemma formalizes.
\end{boundary-case}

% --- Circular Dependency Check ---
\begin{circularity-check}
\textbf{Does Lemma~3.1 depend on any later result?} No. It depends only
on Assumptions~1--2 and the definitions of $\delta_A^{\text{obs}}$,
$\sigma_A^{\text{obs}}$. It does not require Proposition~3.6 (which
provides a tighter bound on $\tilde{\sigma}_A$), nor any dynamical
result.

\textbf{Does any later result depend on Lemma~3.1?} The argument roadmap
(the original manuscript, \S3) lists Step~1 as ``$\sigma_A$ cannot be
inferred from $\delta_A$ alone [Proposition~3.6--3.7]''. Propositions
3.6--3.7 bound estimator error but do not formally prove
non-injectivity. Lemma~3.1 fills this gap and should be referenced
by Step~1. No circularity.
\end{circularity-check}
