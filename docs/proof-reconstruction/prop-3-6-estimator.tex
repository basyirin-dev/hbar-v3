% --- Definitions and Assumptions ---
\begin{definition}[Two-Stage Estimation]
Let $\tilde{\sigma}_A(d,t)$ be the Stage~1 (training-time) proxy computed
by fusing three structural signals:
\begin{itemize}[nosep]
\item \textbf{GCA} (Generalised Compositional Alignment): correlation
  between structural similarity and performance difference,
\item \textbf{RGA} (Relational Generalisation Alignment): rank correlation
  of compositional generalisation patterns,
\item \textbf{AC} (Augmentation Consistency): agreement under compositional
  input transformations.
\end{itemize}
Let $\hat{\sigma}_A(d,t) = \text{Acc}_{\text{OOD}} / \text{Acc}_{\text{ID}}$
be the Stage~2 (evaluation-time) ground-truth estimate. The discrepancy is
\[
\delta_\sigma(d,t) = \tilde{\sigma}_A(d,t) - \hat{\sigma}_A(d,t).
\]
\end{definition}

\begin{definition}[Generative Grammar $G(d)$]
A probabilistic context-free grammar that generates compositional structures
for domain $d$. The recombination distribution $\mathcal{R}(G(d))$ governs
how primitive elements combine into probe samples used in Stage~1 estimation.
\end{definition}

\begin{assumption}[Correct Generative Model]
The generative grammar $G(d)$ correctly models the domain's compositional
structure. The compositional probe batch $\{s_j\}_{j=1}^{n_{\text{batch}}}$
is drawn i.i.d. from the recombination distribution $\mathcal{R}(G(d))$.
\end{assumption}

\begin{assumption}[Bounded Signals]
Each structural signal is bounded in $[0,1]$ and has finite variance.
The fusion weights are non-negative and sum to 1.
\end{assumption}

% --- Proposition Statement ---
\begin{proposition}[Estimator Consistency]
\label{prop:estimator}
Under Assumptions~1--2, the multi-signal fusion estimator satisfies:
\[
\mathbb{E}[|\delta_\sigma|] \leq
\frac{C_1}{\sqrt{n_{\text{batch}}}} + C_2 \cdot \epsilon_{\text{model}},
\]
where $n_{\text{batch}}$ is the number of compositional probe samples,
$\epsilon_{\text{model}}$ is the model misspecification error, and
$C_1, C_2$ are constants depending on the domain parameters.
\end{proposition}

% --- Proof ---
\begin{proof}
We decompose the expected discrepancy into statistical and systematic
components:

\[
\mathbb{E}[|\delta_\sigma|] \leq
\underbrace{\mathbb{E}[|\tilde{\sigma}_A - \mathbb{E}[\tilde{\sigma}_A]|]}_{\text{statistical}}
+ \underbrace{|\mathbb{E}[\tilde{\sigma}_A] - \sigma_A^*|}_{\text{systematic}}
+ \underbrace{|\sigma_A^* - \hat{\sigma}_A|}_{\text{measurement}},
\]
where $\sigma_A^*$ is the true schema coherence. The third term is
absorbed into $\epsilon_{\text{model}}$ as it represents fundamental
unidentifiability.

\paragraph{Step 1: Statistical term — concentration of the fusion estimate.}
The Stage~1 proxy is a weighted combination:
\[
\tilde{\sigma}_A = w_1 \cdot \text{GCA} + w_2 \cdot \text{RGA} + w_3 \cdot \text{AC}.
\]
Each signal is a U-statistic or sample mean computed from the
compositional probe batch.

For the correlation-based signals (GCA, RGA), each is a
degree-$m$ U-statistic:
\[
U = \binom{n}{m}^{-1} \sum_{1 \leq i_1 < \dots < i_m \leq n}
h(X_{i_1}, \dots, X_{i_m}),
\]
where $h$ is a symmetric kernel bounded in $[0,1]$. By Hoeffding's
inequality for U-statistics (Hoeffding, 1948):
\[
\mathbb{P}(|U - \mathbb{E}[U]| \geq t) \leq 2 \exp(-2 \lfloor n/m \rfloor t^2).
\]
For the augmentation consistency AC, which is a sample mean of bounded
indicators, Hoeffding's inequality gives:
\[
\mathbb{P}(|\text{AC} - \mathbb{E}[\text{AC}]| \geq t) \leq 2 \exp(-2 n_{\text{batch}} t^2).
\]
Applying a union bound over the three signals with $t = \varepsilon/3$:
\[
\mathbb{P}(|\tilde{\sigma}_A - \mathbb{E}[\tilde{\sigma}_A]| \geq \varepsilon)
\leq 6 \exp\left(-2 \lfloor n_{\text{batch}} / m \rfloor (\varepsilon/3)^2\right).
\]
Integrating the tail bound gives:
\[
\mathbb{E}[|\tilde{\sigma}_A - \mathbb{E}[\tilde{\sigma}_A]|]
= \int_0^\infty \mathbb{P}(|\tilde{\sigma}_A - \mathbb{E}[\tilde{\sigma}_A]| \geq t) \, dt
\leq \frac{C_1}{\sqrt{n_{\text{batch}}}},
\]
where $C_1 = 3\sqrt{2m}$ (absorbing constants from the Gaussian tail
integral of the Hoeffding bound).

\paragraph{Step 2: Systematic term — model misspecification.}
The systematic bias $|\mathbb{E}[\tilde{\sigma}_A] - \sigma_A^*|$ arises
from two sources:
\begin{enumerate}[nosep]
\item \textbf{Fusion weight misspecification:} The fixed weights
  $(w_1, w_2, w_3)$ may not be optimal for the current domain.
\item \textbf{Grammar approximation error:} $G(d)$ may not capture all
  compositional structure, inducing a bias in the probe distribution.
\end{enumerate}
Both sources are bounded by a domain-specific constant $C_2$ times
the model misspecification error $\epsilon_{\text{model}}$, defined as
the total variation distance between the true recombination distribution
and the grammar-induced distribution.

\paragraph{Step 3: Measurement term — OOD/ID ratio consistency.}
The Stage~2 estimator $\hat{\sigma}_A = \text{Acc}_{\text{OOD}} /
\text{Acc}_{\text{ID}}$ is a ratio of two bounded random variables.
When $\text{Acc}_{\text{ID}} > 0$ (non-degenerate case), the Delta
method gives:
\[
|\sigma_A^* - \hat{\sigma}_A| \leq \frac{|\text{Acc}_{\text{OOD}} - \mathbb{E}[\text{Acc}_{\text{OOD}}]|}{\mathbb{E}[\text{Acc}_{\text{ID}}]}
+ \frac{\mathbb{E}[\text{Acc}_{\text{OOD}}] \cdot |\text{Acc}_{\text{ID}} - \mathbb{E}[\text{Acc}_{\text{ID}}]|}{(\mathbb{E}[\text{Acc}_{\text{ID}}])^2} + o_p(1).
\]
Each accuracy estimate concentrates at rate $1/\sqrt{n_{\text{eval}}}$.
For $n_{\text{eval}} \geq n_{\text{batch}}$, this term is dominated by
the statistical term from Step~1.

\paragraph{Step 4: Combining terms.}
Taking expectations and collecting constants:
\[
\mathbb{E}[|\delta_\sigma|] \leq \frac{3\sqrt{2m}}{\sqrt{n_{\text{batch}}}}
+ C_2 \cdot \epsilon_{\text{model}}.
\]
Setting $C_1 = 3\sqrt{2m}$ yields the stated bound.
$\square$
\end{proof}

% --- Boundary Cases ---
\begin{boundary-case}
\textbf{$n_{\text{batch}} \to \infty$:}
The statistical term vanishes and $\mathbb{E}[|\delta_\sigma|] \to C_2 \epsilon_{\text{model}}$.
Even with infinite data, systematic misspecification persists.

\textbf{$\epsilon_{\text{model}} = 0$:}
If $G(d)$ is exact, $\mathbb{E}[|\delta_\sigma|] \leq C_1 / \sqrt{n_{\text{batch}}}$,
decaying to zero as $n_{\text{batch}} \to \infty$.

\textbf{Acc${}_{\text{ID}} = 0$:}
The ratio $\hat{\sigma}_A$ is undefined (division by zero). This occurs
when the agent has zero accuracy on ID tasks, indicating catastrophic
forgetting. The estimator is undefined and the recalibration rule
triggers a diagnostic.

\textbf{Discrepancy $> 0.15$:}
The recalibration rule (ridge regression refit of fusion weights) is
triggered. The systematic bias term $C_2 \epsilon_{\text{model}}$ is
reduced, at the cost of introducing finite-sample noise from the
smaller recalibration dataset.
\end{boundary-case}

% --- Circular Dependency Check ---
\begin{circularity-check}
\textbf{Dependencies on earlier results:} The proof uses standard
concentration inequalities and does not depend on any earlier
proposition. It assumes the ODE state variables are well-defined
(Section~3.1) but does not require Proposition~3.1 or 3.2.

\textbf{Forward dependencies:} Proposition~3.7 (error propagation)
depends on this proposition to bound $|\delta_\sigma|$. This is a
forward dependency, not circular.

\textbf{Self-circularity:} The bound $C_1 = 3\sqrt{2m}$ requires
$m$, the degree of the U-statistic kernel. For GCA (Pearson
correlation), $m = 2$; for RGA (Spearman rank correlation),
$m = 2$; for AC (sample mean), $m = 1$. A conservative bound
takes $m = 2$, giving $C_1 = 6$. The constant $C_2$ depends on
the domain and is not explicitly computed; its existence follows
from the boundedness of the fusion estimator and the total variation
bound on the grammar approximation error.
\end{circularity-check}
